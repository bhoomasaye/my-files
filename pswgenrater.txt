import random
import string
import tkinter as tk
from tkinter import messagebox

# Function to generate password
def generate_password():
    try:
        length = int(entry.get())
        if length < 1:
            messagebox.showerror("Error", "Enter a number greater than zero!")
            return
        
        characters = string.ascii_letters + string.digits + string.punctuation
        password = ''.join(random.choice(characters) for _ in range(length))
        
        # Display generated password in the output field
        password_output.config(state='normal')  # Make the field editable temporarily
        password_output.delete(0, tk.END)  # Clear any previous password
        password_output.insert(0, password)  # Insert the generated password
        password_output.config(state='readonly')  # Make it readonly again
        
    except ValueError:
        messagebox.showerror("Error", "Please enter a valid number for length")

# Function to copy the password
def copy_password():
    root.clipboard_clear()  # Clear the clipboard
    root.clipboard_append(password_output.get())  # Copy the generated password to clipboard
    messagebox.showinfo("Copied", "Password copied to clipboard!")

# Create the main window
root = tk.Tk()
root.title("Colorful Password Generator")
root.geometry("400x400")
root.config(bg="#f0f8ff")  # Set background color to light blue

# Create a frame for better organization
frame = tk.Frame(root, bg="#f0f8ff")
frame.pack(pady=20)

# Label for password length input
label = tk.Label(frame, text="Enter the length for your password:", font=("Arial", 12, "bold"), bg="#f0f8ff")
label.pack(pady=10)

# Entry widget for user to input password length
entry = tk.Entry(frame, width=25, font=("Arial", 12), bg="#e6e6fa", fg="black")
entry.pack(pady=5)

# Button to generate password with vibrant colors
generate_button = tk.Button(frame, text="Generate Password", font=("Arial", 12), bg="#ffcccb", fg="black", command=generate_password)
generate_button.pack(pady=10)

# Output field to display generated password
password_output = tk.Entry(frame, width=30, font=("Arial", 12), bg="#fafad2", state='readonly')
password_output.pack(pady=5)

# Button to copy the generated password
copy_button = tk.Button(frame, text="Copy Password", font=("Arial", 12), bg="#90ee90", fg="black", command=copy_password)
copy_button.pack(pady=10)

# Add a footer label
footer = tk.Label(root, text="Enjoy your secure password!", font=("Arial", 10, "italic"), bg="#f0f8ff")
footer.pack(pady=10)

# Start the GUI event loop
root.mainloop()
